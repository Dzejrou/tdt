150 void GraphicsHelper::init_graphics_component(EntitySystem& ents, Ogre::SceneManager& scene, tdt::uint id)
151 {
152  auto comp = ents.get_component<GraphicsComponent>(id);
153 
154  if(!comp)
155  return;
156 
157  if(comp->node && comp->entity)
158  {
159  comp->node->detachObject(comp->entity);
160  scene.destroyEntity(comp->entity);
161  }
162 
163  if(!comp->node)
164  comp->node = scene.getRootSceneNode()->createChildSceneNode("entity_" + std::to_string(id));
165 
166  comp->entity = scene.createEntity(comp->mesh);
167  comp->entity->setRenderingDistance(4000.f);
168  comp->node->attachObject(comp->entity);
169  comp->node->setVisible(comp->visible);
170 
171  if(comp->manual_scaling)
172  comp->node->setScale(comp->scale);
173 
174  if(comp->material != "NO_MAT")
175  comp->entity->setMaterialName(comp->material);
176 
177  auto half_height = comp->entity->getWorldBoundingBox(true).getHalfSize().y;
178  auto phys_comp = ents.get_component<PhysicsComponent>(id);
179  if(phys_comp)
180  {
181  phys_comp->half_height = half_height;
182  phys_comp->position = Ogre::Vector3{phys_comp->position.x,
183  half_height, phys_comp->position.z};

185  }
186 
187  auto light = ents.get_component<LightComponent>(id);
188  if(light && light->light)

190  light->node = comp->node;
